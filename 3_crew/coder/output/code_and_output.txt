# Python program to calculate the first 10,000 terms of the series 
# 1 - 1/3 + 1/5 - 1/7 + ... and multiply the total by 4

def calculate_series(n_terms):
    total = 0.0
    for k in range(n_terms):
        # Determine the sign based on the term index
        sign = 1 if k % 2 == 0 else -1
        term = sign / (2 * k + 1)
        total += term
    return total * 4

# Calculate the series for the first 10,000 terms
result = calculate_series(10000)
print(result)

# Output of the program:
# 3.1414926535900345
